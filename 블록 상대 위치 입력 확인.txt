import pygame, sys
from pygame.locals import *
from block import *

FPS = 5
WINWIDTH = 600
WINHEIGHT = 600
BLOCKHEIGAP = 63
BLOCKWIDGAP = 73
WHITE = (255, 255, 255)

BOARD = pygame.image.load('picture/board.png')
SELECTED = pygame.image.load('picture/selected.png')
IMGDICT = {''}
pygame.init()
FPSCLOCK = pygame.time.Clock()
DISPLAYSURF = pygame.display.set_mode((WINWIDTH, WINHEIGHT))
pygame.display.set_caption('남극의 펭귄을 지켜라~!')

DISPLAYSURF.fill(WHITE)
DISPLAYSURF.blit(BOARD, (0, 0))

i = 0
j = 0
while True:
    if j == 6:
        DISPLAYSURF.blit(BOARD, (0, 0))
        i+=1
        j = 0

    if blocks[i].surs[j].num != WALL:
        surblock = blocks[blocks[i].surs[j].num]
        DISPLAYSURF.blit(SELECTED, BLOCKPOSBOX[surblock.num])

    j+=1
    for event in pygame.event.get():
        pos = pygame.mouse.get_pos()

        if event.type == QUIT or (event.type == KEYUP and event.key == K_ESCAPE):
            pygame.quit()
            sys.exit()

    pygame.display.update()
    FPSCLOCK.tick(FPS)
